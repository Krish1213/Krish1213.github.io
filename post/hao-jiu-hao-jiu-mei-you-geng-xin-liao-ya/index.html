function LogicOut=Phase1(LogicIn, LogicRow, n)
    if LogicIn==1
        LogicOut = 1;
        for i=1:n
            if LogicRow(1,i)==0
                LogicOut=0;
                break;
            end
        end
    else
        LogicOut=0;
    end
end
